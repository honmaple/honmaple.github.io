vue分页

需要参数:

  • page: 当前页码
  • total_page: 总页数

代码片段

&lt;ul class="pagination" v-if="total_page &gt; 1"&gt;
  &lt;li  v-if="page &gt; 1" class="previous"&gt;
    &lt;a href="javascript:void(0);" v-on:click="paginate_to(1);"&gt;«&lt;/a&gt;
  &lt;/li&gt;
  &lt;template v-for="p in total_page"&gt;
    &lt;template v-if="p == page"&gt;
      &lt;li class="active"&gt;
        &lt;a href="javascript:void(0);"&gt;{{ p }}&lt;/a&gt;
      &lt;/li&gt;
    &lt;/template&gt;
    &lt;template v-else-if="(page - 2 &lt; p && p &lt; page + 2) || p &lt;= 3 || p &gt; total_page - 3 "&gt;
      &lt;li&gt;
        &lt;a href="javascript:void(0);" v-on:click="paginate_to(p);"&gt;{{ p }}&lt;/a&gt;
      &lt;/li&gt;
    &lt;/template&gt;
    &lt;template v-else-if="(page - 3 &lt; p  && p  &lt; page + 1) || p &lt;= 2 || p &gt; total_page - 4 "&gt;
      &lt;li&gt;
        &lt;a href="javascript:void(0);" v-on:click="paginate_to(p);"&gt;.....&lt;/a&gt;
      &lt;/li&gt;
    &lt;/template&gt;
  &lt;/template&gt;
  &lt;li v-if="page &lt; total_page" class="previous"&gt;
    &lt;a href="javascript:void(0);" v-on:click="paginate_to(total_page);"&gt;»&lt;/a&gt;
  &lt;/li&gt;
&lt;/ul&gt;
