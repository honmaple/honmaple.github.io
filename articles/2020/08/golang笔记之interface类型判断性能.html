 13type A struct {
 14    v int
 15}
 16
 17func (a *A) AA() {
 18    a.v += 1
 19}
 20
 21func BenchmarkTypeSwitch(b *testing.B) {
 22    var a = new(A)
 23
 24    switchFunc := func(v interface{}) {
 25        switch v.(type) {
 26        case InterfaceA:
 27            v.(InterfaceA).AA()
 28        case InterfaceB:
 29            v.(InterfaceB).BB()
 30        }
 31    }
 32    for i := 0; i < b.N; i++ {
 33        switchFunc(a)
 34    }
 35}
