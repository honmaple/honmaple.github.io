2
3_letter_cases = "abcdefghjkmnpqrstuvwxy" # 去除可能干扰的i，l，o，z

干扰线函数

1def create_lines(self,draw,n_line,width,height):
2    line_num = random.randint(n_line[0],n_line[1]) # 干扰线条数
3    for i in range(line_num):
4        # 起始点
5        begin = (random.randint(0, width), random.randint(0, height))
6        #结束点
7        end = (random.randint(0, width), random.randint(0, height))
8        draw.line([begin, end], fill=(0, 0, 0))

3.绘制干扰点
在绘制干扰线后添加

1if draw_points:
2    self.create_points(draw,point_chance,width,height) #绘制干扰点

干扰点函数

1def create_points(self,draw,point_chance,width,height):
2    chance = min(100, max(0, int(point_chance))) # 大小限制在[0, 100]
3
4    for w in range(width):
5        for h in range(height):
6            tmp = random.randint(0, 100)
7            if tmp > 100 - chance:
8                draw.point((w, h), fill=(0, 0, 0))

4.生成验证码

验证码函数

 1def create_strs(self,draw,chars,length,font_type, 
 2                font_size,width,height,fg_color):
 3    c_chars = random.sample(chars, length)
 4    strs = ' %s ' % ' '.join(c_chars) # 每个字符前后以空格隔开
 5
 6    font = ImageFont.truetype(font_type, font_size)
 7    font_width, font_height = font.getsize(strs)
 8
 9    draw.text(((width - font_width) / 3, (height - font_height) / 3),strs,\
10              font=font, fill=fg_color)
