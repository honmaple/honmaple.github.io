python的os模块学习

2016年07月28日 周四 python os

快速导航

  1. Process Parameters
    1. os.environ(环境变量)
    2. os.getenv(key, default=None)
  2. os.path
    1. os.path.abspath(path) #返回绝对路径
    2. os.path.basename(path) #返回文件名
    3. os.path.commonprefix(list) #返回list(多个路径)中，所有path共有的最长的路径。
    4. os.path.dirname(path) #返回文件路径
    5. os.path.exists(path) #路径存在则返回True,路径不存在返回False
    6. os.path.exists('aa.py')
    7. os.path.lexists #路径存在则返回True,路径不存在也返回False
    8. os.path.expanduser(path) #把path中包含的"~"和"~user"转换成用户目录
    9. os.path.expandvars(path) #根据环境变量的值替换path中包含的”(name”和”){name}”
    10. os.path.getatime(path) #返回最后一次进入此path的时间。
    11. os.path.getmtime(path) #返回在此path下最后一次修改的时间。
    12. os.path.getctime(path)
    13. os.path.getsize(path) #返回文件大小，如果文件不存在就返回错误
    14. os.path.isabs(path) #判断是否为绝对路径
    15. os.path.isfile(path) #判断路径是否为文件
    16. os.path.isdir(path) #判断路径是否为目录
    17. os.path.islink(path) #判断路径是否为链接
    18. os.path.ismount(path) #判断路径是否为挂载点（）
    19. os.path.join(path1[, path2[, …]]) #把目录和文件名合成一个路径
    20. os.path.normcase(path) #转换path的大小写和斜杠
    21. os.path.normpath(path) #规范path字符串形式
    22. os.path.realpath(path) #返回path的真实路径
    23. os.path.relpath(path[, start]) #从start开始计算相对路径
    24. os.path.samefile(path1, path2) #判断目录或文件是否相同
    25. os.path.sameopenfile(fp1, fp2) #判断fp1和fp2是否指向同一文件
    26. os.path.samestat(stat1, stat2) #判断stat tuple stat1和stat2是否指向同一个文件
    27. os.path.split(path) #把路径分割成dirname和basename，返回一个元组
    28. os.path.splitdrive(path) #一般用在windows下，返回驱动器名和路径组成的元组
    29. os.path.splitext(path) #分割路径，返回路径名和文件扩展名的元组
    30. os.path.splitunc(path) #把路径分割为加载点与文件
    31. os.path.walk(path, visit, arg) #遍历path，进入每个目录都调用visit函数，visit函数必须有
    32. os.path.supportsunicodefilenames #设置是否支持unicode路径名

os.path

os.path.abspath(path) #返回绝对路径

1>>> os.path.abspath('aa.py')
2'/home/jianglin/test/aa.py'

os.path.basename(path) #返回文件名

1>>> os.path.basename('aa.py')
2'aa.py'

os.path.commonprefix(list) #返回list(多个路径)中，所有path共有的最长的路径。

1>>> os.path.commonprefix(['aa.py','a.py'])
2'a'
3>>> os.path.commonprefix(['aa.py','c.py'])
4''
5>>> os.path.commonprefix(['~/test/aa.py','~/test/c.py'])
6'~/test/'
7>>> os.path.commonprefix(['~/test/aa.py','~/test/c.py','~/test/git/.git/config'])
8'~/test/'

os.path.dirname(path) #返回文件路径

1>>> os.path.dirname('~/test/aa.py')
2'~/test'

os.path.exists(path) #路径存在则返回True,路径不存在返回False
