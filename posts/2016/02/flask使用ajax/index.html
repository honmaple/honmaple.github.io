1var csrftoken = "{{ csrf_token() }}"
2    $.ajaxSetup({
3        beforeSend: function(xhr, settings) {
4            if (!/^(GET|HEAD|OPTIONS|TRACE)$/i.test(settings.type) && !this.crossDomain) {
5                xhr.setRequestHeader("X-CSRFToken", csrftoken)
6            }
7        }
