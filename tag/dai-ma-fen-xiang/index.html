函数版

import random

    min_num = 0
    max_num = 1000
    selected = []


    def random_go(origin, go):
        go = random.randint(min_num + 1, max_num - 1)
        if go in selected:
            return random_go(origin, go)
        selected.append(go)
        return go


    def guess(origin, go):
        global min_num, max_num
        if origin > go:
            print(go, '小了')
            min_num = max(go, min_num)
            return guess(origin, random_go(origin, go))
        elif origin < go:
            print(go, '大了')
            max_num = min(go, max_num)
            return guess(origin, random_go(origin, go))
        else:
            print(go, '恭喜')
            return True

类版

import random

    class GuessNum(object):
        def __init__(self):
            self.max_num = 10000
            self.min_num = 0
            self.selected = []

        def random_go(self, origin, go):
            go = random.randint(self.min_num + 1, self.max_num - 1)
            if go in self.selected:
                return self.random_go(origin, go)
            self.selected.append(go)
            return go

        def guess(self, origin, go):
            if origin > go:
                print(go, '小了')
                self.min_num = max(go, self.min_num)
                return self.guess(origin, self.random_go(origin, go))
            elif origin < go:
                print(go, '大了')
                self.max_num = min(go, self.max_num)
                return self.guess(origin, self.random_go(origin, go))
            else:
                print(go, '恭喜')
                return True

调用

origin = random.randint(0, 1000)
